#!/bin/bash

# =============================================================================
# SOFTWARE LOCATIONS
# Absolute paths to external tools and JAR files used by the pipeline.
# Update these paths to match where each tool is installed on your system.
# All tools must be installed and accessible before running the pipeline.
# =============================================================================


# -----------------------------------------------------------------------------
# RUNTIME
# -----------------------------------------------------------------------------

# Java runtime — required for JAR-based tools (fgbio, picard, beagle)
# Minimum version: Java 8
java="/usr/lib/jvm/java-1.8.0-openjdk-1.8.0.372.b07-2.el9.x86_64/jre/bin/java"


# -----------------------------------------------------------------------------
# READ PROCESSING & QUALITY CONTROL
# -----------------------------------------------------------------------------

# fgbio — toolkit for demultiplexing library-level FASTQ files
# https://fulcrumgenomics.github.io/fgbio/tools/latest/DemuxFastqs.html
fgbio="/tscc/projects/ps-palmer/software/local/src/fgbio-1.3.0/fgbio-1.3.0.jar"

# BBMap — toolkit for adapter trimming and read quality filtering
# https://archive.jgi.doe.gov/data-and-tools/software-tosaols/bbtools/bb-tools-user-guide/bbduk-guide/
bbmap="/tscc/projects/ps-palmer/software/local/src/bbmap"

# FastQC — per-base sequence quality reports for raw FASTQ files
# https://www.bioinformatics.babraham.ac.uk/projects/fastqc
fastqc="/tscc/projects/ps-palmer/software/local/src/FastQC/fastqc"


# -----------------------------------------------------------------------------
# READ ALIGNMENT & QUALITY CONTROL
# -----------------------------------------------------------------------------

# BWA — Burrows-Wheeler Aligner for paired-end short-read alignment to a reference genome
# https://bio-bwa.sourceforge.net/bwa.shtml
bwa="/tscc/projects/ps-palmer/software/local/src/bwa-0.7.17/bwa"

# SAMtools — toolkit for manipulating and reviewing alignments in SAM/BAM format
# https://www.htslib.org/doc/samtools.html
samtools="/tscc/projects/ps-palmer/software/local/src/samtools-1.14/samtools"

# Picard — toolkit for BAM/SAM file duplicate marking
# https://broadinstitute.github.io/picard/command-line-overview.html#MarkDuplicates
picard="/tscc/projects/ps-palmer/software/local/src/picard-2.25.7/picard.jar"


# -----------------------------------------------------------------------------
# VCF FILE MANIPULATION
# -----------------------------------------------------------------------------

# BCFtools — toolkit for VCF/BCF file manipulation
# https://samtools.github.io/bcftools/bcftools.html
bcftools="/tscc/projects/ps-palmer/software/local/src/bcftools-1.14/bcftools"

# BCFtools plugins directory — required for extended BCFtools functionality
# https://samtools.github.io/bcftools/bcftools.html
bcftools_plugins="/tscc/projects/ps-palmer/software/local/src/bcftools-1.14/plugins"


# -----------------------------------------------------------------------------
# POST-GENOTYPING ANALYSIS
# -----------------------------------------------------------------------------

# PLINK 2 — genotype data management and analysis
# https://www.cog-genomics.org/plink/2.0
plink2="/tscc/projects/ps-palmer/software/local/src/plink2"

# PLINK 1.9 — classic version retained for compatibility with legacy steps
# https://www.cog-genomics.org/plink
plink="/tscc/projects/ps-palmer/software/local/src/plink-1.90/plink"
